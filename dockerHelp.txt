    --name patientsystem-backend \
    --network patientsystem-net \

    patientsystem-backend



om man inte gjort det gör först:

docker network create patientsystem-net

docker run -d \
    --name mysql-container \
    --network patientsystem-net \
    -p 3306:3306 \
    -e MYSQL_ROOT_PASSWORD=supersecret_root \
    -e MYSQL_DATABASE=patientsystemdb \
    -e MYSQL_USER=user \
    -e MYSQL_PASSWORD=supersecret \
    -v mysql_data:/var/lib/mysql \
    mysql:8.0


Logga in i docker mysql:
docker exec -it mysql-container mysql -u user -p
